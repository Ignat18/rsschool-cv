def q_sort(arr):
    if len(arr) < 2:
        return arr
    else:
        pivot = arr[0]
        less = [i for i in arr if i < pivot]
        more = [j for j in arr if j > pivot]
        return q_sort(less) + [pivot] + q_sort(more)
